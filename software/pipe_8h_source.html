1010 static constexpr uint32_t kBufMask = (kMaxCapacity + kQuantumSize) - 1;
1011 static_assert((kBufMask & (kBufMask + 1)) == 0,
1012 "(kBufMask + 1) must be a power of 2");
1013
1014 // Buffer layout:
1015 // | app_begin_ | app_end_
1016 // v v
1017 // +---+------------------------+---+---------------------+--------+
1018 // | | Waiting transmission | | Available to user | |
1019 // +---+------------------------+---+---------------------+--------+
1020 // ^ ^
1021 // | hw_begin_ | hw_end_
1022 //
1023 // The area available to the user is between `app_begin_` and `app_end_`.
1024 // The area between `hw_begin_` and `hw_end_` is waiting to be transmitted.
1025 //
1026 // SendAndFree() will advance `app_begin_` by `nb_bytes_`, reducing the
1027 // available space to the user. The size of the available region can be
1028 // obtained by calling `capacity()`.
1029 //
1030 // To reclaim space, the user must call `Extend()`, which will check for
1031 // completions and potentially advance the `hw_begin_`. We can then advance
1032 // `app_end_` to match `hw_begin_`, increasing the available space to the
1033 // user.
1034 //
1035 // All the buffer pointers (i.e., app_begin_, app_end_, hw_begin_, hw_end_)
1036 // are in units of 64 bytes. The buffer itself is a circular buffer, so
1037 // `app_end_` can be smaller than `app_begin_`.
1038 //
1039 // Currently app_begin_ == hw_end_ and app_end_ == hw_begin_. But this may
1040 // change in the future. One reason it might be useful to change this is that
1041 // currently `Device::Send()` blocks when the notification buffer is full. If
1042 // we make sending non-blocking, it may be useful to let the hw_*_ and app_*_
1043 // pointers diverge.
1044};

1278template <typename T>
1279class MessageIteratorBase {
1280 public:
1281 constexpr uint8_t* operator*() { return addr_; }
1282
1283 /*
1284 * This is a bit ugly but, as far as I know, there is no way to check for the
1285 * end of a range-based for loop without overloading the != operator.
1286 *
1287 * The issue is that it does not work as expected for actual inequality check.
1288 */
1289 constexpr bool operator!=(const MessageIteratorBase& other) const {
1290 return (missing_messages_ != 0) && (next_addr_ <= other.addr_);
1291 }
1292
1293 constexpr T& operator++() {
1294 T* child = static_cast<T*>(this);
1295
1296 uint32_t nb_bytes = next_addr_ - addr_;
1297
1298 child->OnAdvanceMessage(nb_bytes);
1299
1300 addr_ = next_addr_;
1301 next_addr_ = child->GetNextMessage(addr_);
1302
1303 --missing_messages_;
1304 batch_->NotifyProcessedBytes(nb_bytes);
1305
1306 return *child;
1307 }
1308
1309 protected:
1310 inline MessageIteratorBase()
1311 : addr_(nullptr),
1312 missing_messages_(0),
1313 batch_(nullptr),
1314 next_addr_(nullptr) {}
